python manage.py runserver\

python3.8 -m venv venv\
source venv/bin/activate\
pip3.8 install -r requirements.txt\
Adicionar DNS a setting em ALLOWED_HOSTS = ['']\
python3.8 manage.py runserver 0.0.0.0:8000\
